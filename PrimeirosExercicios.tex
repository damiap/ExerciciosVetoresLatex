\subsection{Soma e multiplicação por escalar}

\subsubsection{Seja $\vec{A}=(1,3,6)$, $\vec{B}=(4,-3,3)$ e $\vec{C}=(2,1,5)$ três vetores. Determine as componentes de:}

\begin{enumerate}
 \item $A + B$ = 
 (5, 0, 9) 
 
 \item $A - B$ = 
 (-3, 6, 3)
 
 \item $A + B - C$ =>
  $(5, 0, 9) - (2, 1, 5)$(-5, 5, -2)
 
 \item $7A - 2B - 3C$ = 
 (-7, 24, 21)
 
 \item $2A + B - 3C$ => 
 $(6, 3, 15) - (6, 3, 15)$ = (0, 0, 0)

\end{enumerate}
